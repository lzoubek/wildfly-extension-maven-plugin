88  	public void build() throws Exception {
89  		DocumentBuilder dBuilder = factory.newDocumentBuilder();		
90  		Document srcDoc = dBuilder.parse(sourceFile);
91  		
92  		// sort inserts by the shortest 
93  		Collections.sort(getInserts(), new Comparator<Insert>() {
94  
95  			public int compare(Insert o1, Insert o2) {
96  				Integer o1Size = o1.getSelect().split("/").length;
97  				Integer o2Size = o2.getSelect().split("/").length;
98  				return o1Size.compareTo(o2Size);
99  			}
100 		});
101 		debug("Building ["+this.sourceFile+"] ");
102 		// if our target document defines namespace in root element, let's read it
103 		String namespace = getNameSpace(srcDoc);
104 		if (namespace != null) {
105 			namespaceContext.mapping(PREFIX, namespace);	
106 		}
107 		
108 		for (Insert insert : getInserts()) {
109 			debug("Applying "+insert);
110 			String expression = insert.getSelect();
111 			
112 			if (namespace != null) {
113 				// enhance given xpath to use namespaces
114 				expression = xpath2Namespaced(expression, PREFIX);
115 				debug("Expression "+expression);
116 			}
117 			XPathExpression expr = xpath.compile(expression);
118 			try {
119 				NodeList nl = (NodeList)expr.evaluate(srcDoc, XPathConstants.NODESET);
120 				if (nl.getLength() == 0) {
121 					if (failNoMatch) {
122 						throw new Exception("Failed to update ["+targetFile.getAbsolutePath()+"] "+insert+" does not select any element");
123 					}
124 					warning(insert+" does not select any element");
125 					continue;
126 				}
127 				debug("Expression evaluated to "+nl.getLength()+" nodes");
128 				Document contentDoc = null;
129 				if (insert.getContent() != null) {
130 					debug("Loading content XML from file "+insert.getContent());
131 					contentDoc = dBuilder.parse(insert.getContent());	
132 				} else {
133 					debug("Loading content XML from string");
134 					contentDoc = dBuilder.parse(new ByteArrayInputStream(insert.getXml().getBytes()));
135 				}
136 				
137 				for (int i = 0; i< nl.getLength();i++) {
138 					Node node = nl.item(i);
139 					if (node instanceof Element) {
140 						Element element = (Element)node;
141 						Node inserting = contentDoc.getDocumentElement().cloneNode(true);
142 						srcDoc.adoptNode(inserting);
143 						String recentNs = findRecentNamespaceFromXpath(expression);
144 						// is the root node of inserting content already present?
145 
146 						XPathExpression contentExpr = createContentRootExpression(contentDoc, recentNs, namespace, insert.getAttribute());
147 						NodeList existingNodes = (NodeList)contentExpr.evaluate(element, XPathConstants.NODESET);
148 						if (existingNodes.getLength() > 0) {
149 							// we need to remove those? (could be many)
150 							// we'll replace the last guy
151 							element.replaceChild(inserting, existingNodes.item(existingNodes.getLength()-1));
152 						}
153 						else {
154 							element.appendChild(inserting);
155 						}
156 						
157 						String contentNs = getNameSpace(contentDoc);
158 						// find most recent NS from inserted node back to root node and assign to it
159 						recentNs = findRecentNamespace(srcDoc, inserting);
160 						if (contentNs == null && recentNs != null) {
161 							// content document does not have namespace, let's rename it to our namespace
162 							renameNamespaceRecursive(srcDoc, inserting, recentNs);
163 						}
164 					}
165 				}
166 			} catch (Exception e) {
167 				e.printStackTrace();
168 			}
169 		}
170 		writeTargetDcoument(srcDoc);	
171 	}

178 	public static String findRecentNamespaceFromXpath(String expression) {
179 		Matcher m = NS_IN_XPATH.matcher(expression);
180 		String ns = null;
181 		while (m.find()) {
182 			ns = m.group(1);
183 		}
184 		return ns;
185 	}

187 	private String findRecentNamespace(Document doc, Node node) {
188 		Node parent = null;
189 		while ((parent = node.getParentNode()) != null) {			
190 			Element parentEl = (Element)parent;
191 			String ns = parentEl.getAttribute("xmlns");
192 			if (!ns.isEmpty()) {
193 				return ns;
194 			}
195 			if (parent.isSameNode(doc.getDocumentElement())) {
196 				return null;
197 			}
198 			if (parent.isEqualNode(node)) {
199 				return null;
200 			}
201 			node = parent;
202 		}
203 		return null;
204 	}

220 	public static String xpath2Namespaced(String expression, String prefix) {
221 		String[] components = expression.split("/");
222 		StringBuilder sb = new StringBuilder();
223 		for (int i = 0; i< components.length;i++) {
224 			String piece = components[i];
225 			if (piece.matches("^\\w+.*")) {
226 				piece = prefix+":"+piece;
227 			}
228 			sb.append(piece+"/");
229 		}
230 		if (sb.length()> 0) {
231 			sb.deleteCharAt(sb.length()-1);	
232 		}		
233 		return sb.toString();
234 	}

238 	public static String element2Xpath(Element element, String prefix, String defaultPrefix, String identityAttribute) {
239 		StringBuilder sb = new StringBuilder("/");
240 		String ns = element.getAttribute("xmlns");
241 		String pref = "";
242 		if (!ns.isEmpty()) {
243 			pref = prefix+":";
244 		} else if (defaultPrefix != null) {
245 			pref = defaultPrefix+":";
246 		}
247 		sb.append(pref+element.getLocalName());
248 		if (identityAttribute != null && !identityAttribute.isEmpty()) {
249 			sb.append("[@"+identityAttribute+"='"+element.getAttribute(identityAttribute)+"']");
250 		} else {
251 			// use all attributes
252 			NamedNodeMap attributes = element.getAttributes();
253 			if (attributes.getLength() > 0) {
254 				StringBuilder sbAttributes = new StringBuilder();
255 				sbAttributes.append("[");
256 				int validAttributes = 0;
257 				for (int i=0; i<attributes.getLength(); i++) {
258 					Node node = attributes.item(i);
259 					if ("xmlns".equals(node.getNodeName())) {
260 						// ignore this guy as it's handled with prefixing
261 						continue;
262 					}
263 					validAttributes++;
264 					sbAttributes.append("@"+node.getNodeName()+"='"+node.getNodeValue()+"' and ");
265 				}
266 				if (validAttributes > 0) {
267 					sbAttributes.delete(sbAttributes.length()-5,sbAttributes.length()); // delete last ' and '
268 					sbAttributes.append("]");
269 					sb.append(sbAttributes.toString());
270 				}
271 			}
272 		}
273 		
274 		return sb.toString();
275 	}

277 	private XPathExpression createContentRootExpression(Document contentDoc, String contentNamespace, String rootNamespace, String identityAttribute) throws Exception {
278 		String expression = null;
279 
280 		if (contentNamespace != null) {
281 			expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, PREFIX_CONTENT, identityAttribute);
282 			namespaceContext.mapping(PREFIX_CONTENT, contentNamespace);
283 		} else {
284 			String ns = getNameSpace(contentDoc);
285 			if (ns != null) {
286 				expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, null, identityAttribute);
287 				namespaceContext.mapping(PREFIX_CONTENT, ns);
288 			} else {
289 				expression = element2Xpath(contentDoc.getDocumentElement(), PREFIX_CONTENT, rootNamespace == null ? null : PREFIX, identityAttribute);
290 			}
291 		}
292 		
293 		debug("Content expression "+expression);
294 		// our expression always starts with /, but we'll be evaluating it in context of some other node, thus it needs to be relative		
295 		return xpath.compile(expression.substring(1));
296 	}

360 		File f = insert.getContent();
361 		if (f == null && insert.getXml() == null) {
362 			throw new IllegalArgumentException("Either content or xml must be specified");
363 		}
364 		if (f !=null && !(f.isFile() && f.canRead())) {
365 			throw new IllegalArgumentException("File ["+f.getAbsolutePath()+"] is not a readable file");
366 		}
367 		String expression = insert.getSelect();
368 		if (expression.length() > 1 && expression.endsWith("/")) { // remove trailing slash
369 			insert.setSelect(expression.substring(0, expression.length()-1));
370 		}
371 		try {
372 			xpath.compile(insert.getSelect());	
373 		} catch (XPathExpressionException xee) {
374 			throw new XPathExpressionException(insert.getSelect()+" is not a valid xpath : "+xee.getMessage());
375 		}

394 	private static void renameNamespaceRecursive(Document doc, Node node, String namespace) {
395 		if (node.getNodeType() == Node.ELEMENT_NODE) {
396 			Element el = (Element)node;
397 			if (el.getAttribute("xmlns").isEmpty()) {
398 				doc.renameNode(node, namespace, node.getNodeName());	
399 			} else {
400 				return; // stop renaming ns here, since all children of this node belong to it's ns
401 			}
402 			
403 		}
404 
405 		NodeList list = node.getChildNodes();
406 		for (int i = 0; i < list.getLength(); ++i) {
407 			renameNamespaceRecursive(doc, list.item(i), namespace);
408 		}
409 	}
