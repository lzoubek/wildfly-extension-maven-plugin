55  	/**
56  	 * registers extension to standalone.xml
57  	 * @param options
58  	 * @param destFile output file (standalone.xml or domain.xml with new stuff)
59  	 * @param moduleId ID of JBoss Module to register as an JBoss extension
60  	 * @throws Exception
61  	 */
62  	public void register(RegisterOptions options, File destFile, String moduleId) throws Exception {
63  		try {
64  			log.info("Register extension module="+moduleId);
65  			DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
66  			DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
67  			
68  			// we still need to read standalone.xml and subsystem.xml to DOM to be able to detect namespaces
69  			Document srcDoc = dBuilder.parse(options.getServerConfig());
70  			Document subsystemDoc = null;
71  			Document socketBindingDoc = null;
72  
73  			// unset profiles in case we're not editing domain.xml
74  			if (isStandaloneXml(srcDoc)) {
75  				options.profiles(null);
76  			}
77  			
78  			if (options.getSubsystem().isFile() && options.getSubsystem().canRead()) {
79  				subsystemDoc = dBuilder.parse(options.getSubsystem());
80  				log.info("Configuring subsystem from file "+options.getSubsystem());
81  			} else {
82  				log.info("Subsystem file ["+options.getSubsystem()+"] does not exist, subsystem will not be configured");
83  			}
84  			
85  			if (options.getSocketBinding().isFile() && options.getSocketBinding().canRead()) {
86  				socketBindingDoc = dBuilder.parse(options.getSocketBinding());
87  				log.info("Configuring socket-binding from file "+options.getSocketBinding());
88  			} else {
89  				log.info("Socket-binding file ["+options.getSocketBinding()+"] does not exist, socket-binding will not be configured");
90  			}
91  
92  			// do XSL transformation
93  			TransformerFactory f = TransformerFactory.newInstance();
94  			
95  			InputStream stylesheet = createStylesheet(
96  					getNameSpace(srcDoc), 
97  					moduleId, 
98  					subsystemDoc,
99  					options.getProfiles(),
100 					socketBindingDoc,
101 					options.getSocketBindingGroups()
102 			);
103 			
104 			Transformer t = f.newTransformer(new StreamSource(stylesheet));
105 			Source s = new StreamSource(options.getServerConfig());
106 			Result r = new StreamResult(destFile);
107 			t.setURIResolver(null);
108 			t.setOutputProperty(OutputKeys.INDENT, "yes");
109 			t.setOutputProperty(OutputKeys.STANDALONE, "yes");
110 			t.setOutputProperty(OutputKeys.METHOD, "xml");
111 			t.transform(s, r);
112 			log.info("New serverConfig file written to ["+destFile.getAbsolutePath()+"]");
113 
114 		} catch (TransformerConfigurationException e) {
115 			throw new Exception(e.toString());
116 		} catch (TransformerException e) {
117 			throw new Exception(e.toString());
118 		}
119 	}

121 	private InputStream createStylesheet(String namespace, String module, Document subsystem, String[] profileNames, Document socketBinding, String[] socketBindingGroups) throws IOException {
122 		// when writing new nodes we'll ignore wring xmlns attribute
123 		String ignoreNs = "s"; // s is main namespace for server config
124 		String subns = "";
125 		boolean isSubsystem = subsystem != null;
126 		boolean isSocketBinding = socketBinding != null;
127 
128 		StringBuilder sheet = new StringBuilder("<?xml version=\"1.0\" ?>");
129 		sheet.append("<xsl:stylesheet xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\""
130 				+ " xmlns:s=\"" + namespace + "\"");
131 		
132 		if (isSubsystem) {
133 			subns = getNameSpace(subsystem);
134 			sheet.append(" xmlns:sub=\"" + subns + "\"");
135 			ignoreNs += " sub"; // sub is namespace for subsystem we're setting up
136 		}
137 		sheet.append(" exclude-result-prefixes=\"" + ignoreNs
138 				+ "\" version=\"1.0\">");
139 
140 		// for both extensions and profiles we first copy all nodes by excluding potentially existing one and then we append new content
141 
142 		// append new extension
143 		sheet.append("<xsl:variable name=\"extension\"><extension module=\""+ module+ "\" /></xsl:variable>");
144 		sheet.append("<xsl:template match=\"s:extensions\"><xsl:copy><xsl:apply-templates select=\"*[not(@module='"
145 				+ module
146 				+ "')]|node()\" /></xsl:copy>"
147 				+ "</xsl:template><xsl:template match=\"s:extension[last()]\"><xsl:copy-of select=\"$extension\" /></xsl:template>");
148 
149 		// append new subsystem
150 		if (isSubsystem) {
151 			String profileSelector = createXPathNameAttributeSelector(profileNames);
152 			sheet.append("<xsl:variable name=\"subsystem\">"+ doc2string(subsystem)+ "</xsl:variable>");
153 			sheet.append("<xsl:template match=\"@s:profile"+profileSelector+"\"><xsl:copy><xsl:apply-templates select=\"*[not(namespace-uri() ='"
154 					+ subns
155 					+ "')]|node()\" /></xsl:copy>"
156 					+ "</xsl:template><xsl:template match=\"s:profile"+profileSelector+"/*[last()]\"><xsl:copy-of select=\"$subsystem\" /></xsl:template>");
157 		}
158 		
159 		// append new socket-binding
160 		if (isSocketBinding) {
161 			String sbgSelector = createXPathNameAttributeSelector(socketBindingGroups);
162 			String bindingName = socketBinding.getDocumentElement().getAttribute("name");
163 			sheet.append("<xsl:variable name=\"socketBinding\">"+ doc2string(socketBinding)+ "</xsl:variable>");
164 			sheet.append("<xsl:template match=\"@s:socket-binding-group"+sbgSelector+"\"><xsl:copy><xsl:apply-templates select=\"*[not(@name='"
165 					+ bindingName
166 					+ "')]|node()\" /></xsl:copy>"
167 					+ "</xsl:template><xsl:template match=\"s:socket-binding-group"+sbgSelector+"/*[last()]\"><xsl:copy-of select=\"$socketBinding\" /></xsl:template>");
168 		}
169 
170 		// generic identity template
171 		sheet.append("<xsl:template match=\"@*|node()\"><xsl:copy><xsl:apply-templates select=\"@*|node()\" /></xsl:copy></xsl:template>");
172 		sheet.append("</xsl:stylesheet>");
173 		log.debug("XSL :"+sheet);
174 		return new ByteArrayInputStream(IOUtil.toByteArray(sheet.toString()));
175 	}

177 	private static String createXPathNameAttributeSelector(String[] names) {
178 		if (names == null || names.length == 0) {
179 			return "";
180 		}
181 		StringBuilder sb = new StringBuilder("[");
182 		for (String name : names) {
183 			sb.append("@name='"+name+"' or ");
184 		}
185 		sb.delete(sb.length()-4, sb.length());
186 		sb.append("]");
187 		return sb.toString();
188 	}

190 	private static boolean isStandaloneXml(Document doc) {
191 		return "server".equals(doc.getDocumentElement().getNodeName());
192 	}

194 	/**
195 	 * return XML namespace for root of given document
196 	 * @param doc
197 	 * @return
198 	 */
199 	private static String getNameSpace(Document doc) {
200 		if (doc == null) {
201 			return null;
202 		}
203 		return doc.getDocumentElement().getAttribute("xmlns");
204 	}

206 	/**
207 	 * return content of XML document as string
208 	 * @param doc
209 	 * @return
210 	 */
211 	private static String doc2string(Document doc) {
212 		if (doc == null) {
213 			return null;
214 		}
215 		try {
216 			StringWriter sw = new StringWriter();
217 			TransformerFactory tf = TransformerFactory.newInstance();
218 			Transformer transformer = tf.newTransformer();
219 			transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION,
220 					"yes");
221 			transformer.setOutputProperty(OutputKeys.METHOD, "xml");
222 			transformer.setOutputProperty(OutputKeys.INDENT, "yes");
223 			transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
224 			transformer.transform(new DOMSource(doc), new StreamResult(sw));
225 			return sw.toString();
226 		} catch (Exception ex) {
227 			throw new RuntimeException("Error converting to String", ex);
228 		}
229 	}
